public List<List<Integer>> threeSum(int[] nums) {
    List<List<Integer>> res = new ArrayList<>(); 
    if(nums == null || nums.length < 3) return res;
    Arrays.sort(nums);
    for(int i = 0;i<nums.length;i++){

        while(L<R){
            int temp = nums[i] + nums[L] + nums[R];
            if(temp == 0){

                list.add(nums[L]);
                list.add(nums[R]);
